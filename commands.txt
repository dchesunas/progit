git init - инициализация локального репозитория в текущей директории ( .git папку )

git clone url [new name] - клонирование репозитория

git status - вывод статуса репозитория в полном виде ( ?? - untracked, M - modified, A - staged, MM - modified, staged, modified)
-s - вывод статуса репозитория в кратком виде

git add - наложение версионного контроля, добавление изменненого файла в индекс, указание на решенный конфликт в файле

git diff - сравнивает индекс и непроиндексированные файлы
--staged - сравнивает последний коммит и индекс.


git commit - сделать снимок текущего индекса и создать коммит, все недобавленное в индекс остается висеть в репозитории
-a - игнорирование индекса ( автоматически add)

git rm - удаление файла из отслеживаемых (индекс и каталога)
-f - если файл был изменен и проиндексирован, но незакомичен
-- cached - позволяет удалить файл из индекса, но оставить его в каталоге ( например для удаления лишних файлов, проиндексированные до .gitignore)

git mv - переименовывание файлов аналог цепочки ($ mv README.md README
                                                 $ git rm README.md
                                                 $ git add README)

git log - просмотр истории комитов ( новые в верху )
-p - показывает diff
-2 - количество коммитов
--stat - добавляет статистику о комите
--pretty - изменяет вывод в косоль (full, short, fuller, format - для вывода с данными по шаблону)
--since - установка даты, с которой показывать комиты

git remote - просмотр удаленные репозиториев
-v - с урлами

git remote show [remote-name] - получаем полную информация о ваших локальных ветка относительно удаленных и просто о удаленном репозитории

git remote add [shortname] [url] - добавление нового удаленного репозитория

git remote rename [old] [new]

git remote rm [remote-name]

git fetch [remote-name] - забирает изменения с удаленного сервера, которых у вас нет, но ничего не мержит

git pull origin [branch] - забирает изменения и мержит с вашей локально веткой branch

git push [remote-name] [branch-name] - выгрузка изменения на удаленный репозиторий.


git tag -a - создает аннотированую метку с кучей данных

git tag - обычная метка

git push --tags - пуш всех меток


Отмена действий

git commit
--amend - позволяет дополнить предыдущий комит, который еще не был запушен. Просто возьмет обновление индекса и откроется окно с запросом на сообщение в коммите.

git reset HEAD <file> - откат измений, которые попали в индекс.

git checkout -- <file> - откат изменений не проиндексированого файла.
